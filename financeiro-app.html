function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action || (e.postData ? JSON.parse(e.postData.contents).action : ''); var data = e.postData ? JSON.parse(e.postData.contents) : {}; var result = {}; if (action === 'getResumo') { result = getResumo(ss, data.mes, data.ano); } else if (action === 'addEntrada') { result = addEntrada(ss, data); } else if (action === 'addSaida') { result = addSaida(ss, data); } else if (action === 'addParcela') { result = addParcela(ss, data); } else if (action === 'getParcelas') { result = getParcelas(ss, data.mes); } else if (action === 'getInvestimentos') { result = getInvestimentos(ss); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getResumo(ss, mes, ano) { var wsIn = ss.getSheetByName('ENTRADAS'); var wsOut = ss.getSheetByName('SAÍDAS'); var entradas = 0, saidas = 0; var cats = {}; var dataIn = wsIn.getRange(4,1,300,8).getValues(); dataIn.forEach(function(r) { if (r[1] instanceof Date) { if (r[1].getMonth()+1 == mes && r[1].getFullYear() == ano) { entradas += r[5] || 0; } } }); var dataOut = wsOut.getRange(4,1,300,9).getValues(); dataOut.forEach(function(r) { if (r[1] instanceof Date) { if (r[1].getMonth()+1 == mes && r[1].getFullYear() == ano) { var val = r[5] || 0; saidas += val; var cat = r[3] || 'Outros'; cats[cat] = (cats[cat] || 0) + val; } } }); return {entradas:entradas, saidas:saidas, saldo:entradas-saidas, categorias:cats}; } function addEntrada(ss, d) { var ws = ss.getSheetByName('ENTRADAS'); var last = ws.getLastRow() + 1; ws.getRange(last,2).setValue(new Date(d.data)); ws.getRange(last,3).setValue(d.descricao); ws.getRange(last,4).setValue(d.categoria); ws.getRange(last,5).setValue(d.forma); ws.getRange(last,6).setValue(d.valor); ws.getRange(last,7).setValue(d.status || 'Pago'); return {ok:true}; } function addSaida(ss, d) { var ws = ss.getSheetByName('SAÍDAS'); var last = ws.getLastRow() + 1; ws.getRange(last,2).setValue(new Date(d.data)); ws.getRange(last,3).setValue(d.descricao); ws.getRange(last,4).setValue(d.categoria); ws.getRange(last,5).setValue(d.forma); ws.getRange(last,6).setValue(d.valor); ws.getRange(last,8).setValue(d.status || 'Pago'); return {ok:true}; } function addParcela(ss, d) { var ws = ss.getSheetByName('PARCELAMENTOS'); var last = ws.getLastRow() + 1; ws.getRange(last,2).setValue(d.descricao); ws.getRange(last,3).setValue(d.categoria); ws.getRange(last,4).setValue(d.cartao); ws.getRange(last,5).setValue(new Date(d.data)); ws.getRange(last,6).setValue(d.total); ws.getRange(last,7).setValue(d.parcelas); return {ok:true}; } function getParcelas(ss, mes) { var ws = ss.getSheetByName('PARCELAMENTOS'); var data = ws.getRange(5,1,30,11).getValues(); var result = []; data.forEach(function(r) { if (r[1] && r[6] > 0) { result.push({ desc: r[1], cat: r[2], cartao: r[3], dataInicio: r[4], total: r[5], nParcelas: r[6], pagas: r[8] || 0, valorParc: r[5]/r[6] }); } }); return {parcelas: result}; } function getInvestimentos(ss) { var ws = ss.getSheetByName('INVESTIMENTOS'); var caixinhas = ws.getRange(4,1,6,8).getValues(); var btc = ws.getRange(13,1,8,8).getValues(); return {caixinhas: caixinhas, btc: btc}; }
